import React from 'react';
 
export function makeOptions(labels) {
  return [
    { key: '', value: '', label: '(all)' },
  ].concat(Object.keys(labels).map(
    value => ({ value, label: labels[value] }),
  )).map(({ value, label }) => (
    <option key={value} value={value}>{label}</option>
  ));
}
